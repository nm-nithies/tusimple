import cv2
import numpy as np

# Step 1: Read the image
image = cv2.imread('path/to/image.jpg')  # Replace with your image path

# Step 2: Resize the image to 256x256
image = cv2.resize(image, (256, 256))

# Step 3: Convert from BGR to RGB (if needed)
image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)

# Step 4: Transpose to CHW format (Channels, Height, Width)
image = np.transpose(image, (2, 0, 1))

# Step 5: Add a batch dimension to make it (1, 3, 256, 256)
image = np.expand_dims(image, axis=0)

# Verify the shape
print(image.shape)  # Output: (1, 3, 256, 256)
